```python
# backend_generalist/sandbox.py

from pathlib import Path
import subprocess

class SandboxEscapeError(ValueError):
    """Raised when a candidate path or command would escape the session root."""

def resolve_in_sandbox(session_root: Path, candidate: str | Path) -> Path:
    """Resolve `candidate` against `session_root` and confirm it stays inside.

    Resolves with strict=False (target file may not yet exist for Write).
    Rejects:
      - absolute paths not under session_root
      - relative paths whose resolved form escapes session_root via `..`
      - paths whose resolved form is a symlink target outside session_root
    Returns the absolute, resolved Path (always inside session_root).
    Raises SandboxEscapeError on any escape.
    """

def run_bash_in_sandbox(
    session_root: Path,
    command: str,
    timeout: float = 120.0,
) -> subprocess.CompletedProcess[str]:
    """Run `command` via `bash -c` with cwd=session_root.

    Captures stdout+stderr as text. Raises SandboxEscapeError if session_root
    itself is not a directory. Does NOT attempt to whitelist commands —
    full unrestricted Bash is an explicit project decision (HARN-03 / PROJECT.md).
    """
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Agent → Filesystem | The pydantic-ai agent generates path arguments (Read/Write/Edit/Glob); these are untrusted input that must be confined to the session root before any open()/glob()/write_text(). |
| Agent → Shell | The agent generates Bash commands; user explicitly chose unrestricted Bash for harness fidelity. The only mitigation is cwd=session_root, not command whitelisting. |
| User → Session ID | Plan 02 (next plan) generates session IDs; predictable IDs aren't directly exploitable in single-user CLI mode but are tracked here for hand-off. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Tampering | Agent crafts `../../etc/passwd` Read arg | mitigate | `resolve_in_sandbox` rejects via Path.resolve()+containment check; Test 3 enforces this |
| T-01-02 | Tampering | Agent crafts absolute path `/etc/passwd` Read arg | mitigate | `resolve_in_sandbox` rejects via containment check; Test 4 enforces this |
| T-01-03 | Tampering | Agent symlinks `link -> /tmp` then Reads `link/foo` | mitigate | `Path.resolve()` follows the symlink, containment check then rejects; Test 5 enforces this |
| T-01-04 | EoP | Agent writes outside session root via Write | mitigate | All Write paths go through `resolve_in_sandbox`; Test 6 covers non-existent target case |
| T-01-05 | EoP | Agent runs `cd /etc && cat passwd` via Bash | accept | User explicit decision: full unrestricted Bash for max harness fidelity (PROJECT.md, CLAUDE.md). cwd=session_root is the only mitigation; agent CAN escape via `cd`. Documented as accepted risk in STATE.md "Bash misuse — accepted risk". |
| T-01-06 | DoS | Agent runs `sleep 99999` via Bash | mitigate | `run_bash_in_sandbox` enforces `timeout=120s` default; Test 9 enforces TimeoutExpired propagation |
| T-01-07 | Info Disclosure | Sandbox helper logs full path including session UUID at INFO level | accept | Single-user local CLI; no log shipping; benign for v1 |
| T-01-08 | Spoofing | Agent crafts session ID to collide with another session | accept | Session ID generation lives in Plan 02 (uuid4 — collision-resistant); not in scope here |
