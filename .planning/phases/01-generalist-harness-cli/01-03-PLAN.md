    ```python
    """Generic coding-agent tools — Read, Write, Edit, Glob, Bash.

    EXACTLY these five tools are registered on the agent. No domain tools.
    All filesystem args route through resolve_in_sandbox; Bash runs through
    run_bash_in_sandbox. The session_root is carried via RunContext[GMDeps].
    """
    from __future__ import annotations
    from pydantic_ai import RunContext, ModelRetry

    from backend_generalist.sandbox import (
        resolve_in_sandbox,
        run_bash_in_sandbox,
        SandboxEscapeError,
    )

    MAX_BASH_OUTPUT_CHARS = 32_000

    def read_file(ctx: RunContext, path: str) -> str:
        """Read a UTF-8 text file from the session world directory.

        Args:
            path: Relative path inside the session world directory.
        Returns:
            The file's text content.
        """
        try:
            target = resolve_in_sandbox(ctx.deps.session_root, path)
        except SandboxEscapeError as e:
            raise ModelRetry(f"Path '{path}' escapes the session world directory: {e}")
        if not target.is_file():
            raise ModelRetry(f"File not found: {path}")
        return target.read_text(encoding="utf-8")

    def write_file(ctx: RunContext, path: str, content: str) -> str:
        """Write content to a file in the session world directory (creates parent dirs).

        Args:
            path: Relative path inside the session world directory.
            content: Text content to write (overwrites any existing content).
        Returns:
            Confirmation string.
        """
        try:
            target = resolve_in_sandbox(ctx.deps.session_root, path)
        except SandboxEscapeError as e:
            raise ModelRetry(f"Path '{path}' escapes the session world directory: {e}")
        target.parent.mkdir(parents=True, exist_ok=True)
        target.write_text(content, encoding="utf-8")
        return f"Wrote {len(content)} bytes to {path}"

    def edit_file(ctx: RunContext, path: str, old: str, new: str) -> str:
        """Replace ONE occurrence of `old` with `new` in the named file.

        Raises ModelRetry if `old` is not found or appears multiple times.
        """
        try:
            target = resolve_in_sandbox(ctx.deps.session_root, path)
        except SandboxEscapeError as e:
            raise ModelRetry(f"Path '{path}' escapes the session world directory: {e}")
        if not target.is_file():
            raise ModelRetry(f"File not found: {path}")
        text = target.read_text(encoding="utf-8")
        count = text.count(old)
        if count == 0:
            raise ModelRetry(f"Pattern not found in {path}: {old!r}")
        if count > 1:
            raise ModelRetry(
                f"Pattern occurs {count} times in {path}; provide more unique surrounding context."
            )
        target.write_text(text.replace(old, new, 1), encoding="utf-8")
        return f"Edited {path}: 1 replacement"

    def glob_files(ctx: RunContext, pattern: str) -> str:
        """Glob the session world directory for files matching pattern.

        Args:
            pattern: glob pattern (e.g. '**/*.json', 'world/*.json'). Must be relative.
        """
        # Reject patterns that try to escape via leading `/` or `..`
        if pattern.startswith("/") or ".." in pattern.split("/"):
            raise ModelRetry(
                f"Glob pattern '{pattern}' must be relative and within the session dir."
            )
        root = ctx.deps.session_root
        matches = sorted(root.glob(pattern))
        # Filter to entries still inside the sandbox (defensive — symlinks etc.)
        safe: list[str] = []
        for m in matches:
            try:
                resolved = m.resolve()
                resolved.relative_to(root.resolve())
            except (ValueError, OSError):
                continue
            safe.append(str(m.relative_to(root)))
        if not safe:
            return f"No matches for pattern: {pattern}"
        return "\n".join(safe)

    def bash(ctx: RunContext, command: str, timeout: float = 120.0) -> str:
        """Run an unrestricted Bash command with cwd = session world directory.

        Args:
            command: Shell command line (passed to `bash -c`).
            timeout: Seconds before timeout (default 120).
        Returns:
            Combined stdout+stderr output (truncated if very large), with the exit code.
        """
        try:
            result = run_bash_in_sandbox(ctx.deps.session_root, command, timeout=timeout)
        except SandboxEscapeError as e:
            raise ModelRetry(f"Bash sandbox error: {e}")
        out = (result.stdout or "") + (result.stderr or "")
        if len(out) > MAX_BASH_OUTPUT_CHARS:
            out = out[:MAX_BASH_OUTPUT_CHARS] + "\n...[truncated]"
        return f"exit_code={result.returncode}\n{out}"


    def register_tools(agent) -> None:
        """Register the 5 tools on the given pydantic-ai Agent."""
        agent.tool(read_file)
        agent.tool(write_file)
        agent.tool(edit_file)
        agent.tool(glob_files)
        agent.tool(bash)
    ```

    ```python
    Your working directory is the player's WORLD — a directory of JSON and Markdown files
    that ARE the live game state. You have these tools:

    - read_file(path): Read any file inside the world directory.
    - write_file(path, content): Create or overwrite a file. Use to write a brand-new

    - edit_file(path, old, new): Replace one snippet inside an existing file. Use this

    1. Read README.md, pc.json, campaign/index.md, world/scene.json, and rules/core.md.
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Agent → Tool args | The model generates tool args (paths, commands) — these are untrusted. Sandbox primitive (Plan 01) is the only mitigation. |
| Agent → System prompt | The system prompt is trusted (committed to repo); not vulnerable to prompt injection from filesystem because filesystem content is presented as tool output, not as instructions. |
| Player input → Agent | Player chat input flows in via stdin (Plan 04). Player can attempt prompt injection but cannot escalate beyond what the agent already can do (which is bounded by the sandbox). |
